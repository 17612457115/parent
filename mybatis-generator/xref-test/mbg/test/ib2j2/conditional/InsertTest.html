46      public void testFieldsOnlyInsert() {
47          FieldsonlyDAO dao = getFieldsonlyDAO();
48      
49          try {
50              Fieldsonly record = new Fieldsonly();
51              record.setDoublefield(new Double(11.22));
52              record.setFloatfield(new Double(33.44));
53              record.setIntegerfield(new Integer(5));
54              dao.insert(record);
55      
56              FieldsonlyExample example = new FieldsonlyExample();
57              example.createCriteria().andIntegerfieldEqualTo(new Integer(5));
58      
59              List answer = dao.selectByExample(example);
60              assertEquals(1, answer.size());
61      
62              Fieldsonly returnedRecord = (Fieldsonly) answer.get(0);
63              assertEquals(record.getIntegerfield(), returnedRecord
64                      .getIntegerfield());
65              assertEquals(record.getDoublefield(), returnedRecord
66                      .getDoublefield());
67              assertEquals(record.getFloatfield(), returnedRecord.getFloatfield());
68          } catch (Exception e) {
69              fail(e.getMessage());
70          }
71      }

94      public void testPKFieldsInsert() {

97          try {
98              Pkfields record = new Pkfields();
99              record.setDatefield(new Date());
100             record.setDecimal100field(new Long(10L));
101             record.setDecimal155field(new BigDecimal("15.12345"));
102             record.setDecimal30field(new Short((short) 3));
103             record.setDecimal60field(new Integer(6));
104             record.setFirstname("Jeff");
105             record.setId1(new Integer(1));
106             record.setId2(new Integer(2));
107             record.setLastname("Butler");
108             record.setTimefield(new Date());
109             record.setTimestampfield(new Date());
110     
111             dao.insert(record);
112     
113             PkfieldsKey key = new PkfieldsKey();
114             key.setId1(new Integer(1));
115             key.setId2(new Integer(2));
116     
117             Pkfields returnedRecord = dao.selectByPrimaryKey(key);
118             assertNotNull(returnedRecord);
119     
120             assertTrue(TestUtilities.datesAreEqual(record.getDatefield(), returnedRecord
121                     .getDatefield()));
122             assertEquals(record.getDecimal100field(), returnedRecord
123                     .getDecimal100field());
124             assertEquals(record.getDecimal155field(), returnedRecord
125                     .getDecimal155field());
126             assertEquals(record.getDecimal30field(), returnedRecord
127                     .getDecimal30field());
128             assertEquals(record.getDecimal60field(), returnedRecord
129                     .getDecimal60field());
130             assertEquals(record.getFirstname(), returnedRecord.getFirstname());
131             assertEquals(record.getId1(), returnedRecord.getId1());
132             assertEquals(record.getId2(), returnedRecord.getId2());
133             assertEquals(record.getLastname(), returnedRecord.getLastname());
134             assertTrue(TestUtilities.timesAreEqual(record.getTimefield(), returnedRecord
135                     .getTimefield()));
136             assertEquals(record.getTimestampfield(), returnedRecord
137                     .getTimestampfield());

139             fail(e.getMessage());

226     
227         try {
228             AwfulTable record = new AwfulTable();
229     
230             record.seteMail("fred@fred.com");
231             record.setEmailaddress("alsofred@fred.com");
232             record.setFirstFirstName("fred1");
233             record.setFrom("from field");
234             record.setId1(new Integer(1));
235             record.setId2(new Integer(2));
236             record.setId5(new Integer(5));
237             record.setId6(new Integer(6));
238             record.setId7(new Integer(7));
239             record.setSecondFirstName("fred2");
240             record.setThirdFirstName("fred3");
241             record.setActive(true);
242     
243             Integer generatedCustomerId = dao.insert(record);
244             assertEquals(57, generatedCustomerId.intValue());
245     
246             AwfulTable returnedRecord = dao
247                     .selectByPrimaryKey(generatedCustomerId);
248     
249             assertEquals(generatedCustomerId, returnedRecord.getCustomerId());
250             assertEquals(record.geteMail(), returnedRecord.geteMail());
251             assertEquals(record.getEmailaddress(), returnedRecord
252                     .getEmailaddress());
253             assertEquals(record.getFirstFirstName(), returnedRecord
254                     .getFirstFirstName());
255             assertEquals(record.getFrom(), returnedRecord.getFrom());
256             assertEquals(record.getId1(), returnedRecord.getId1());
257             assertEquals(record.getId2(), returnedRecord.getId2());
258             assertEquals(record.getId5(), returnedRecord.getId5());
259             assertEquals(record.getId6(), returnedRecord.getId6());
260             assertEquals(record.getId7(), returnedRecord.getId7());
261             assertEquals(record.getSecondFirstName(), returnedRecord
262                     .getSecondFirstName());
263             assertEquals(record.getThirdFirstName(), returnedRecord
264                     .getThirdFirstName());
265             assertEquals(record.isActive(), returnedRecord
266                     .isActive());
267         } catch (Exception e) {
268             fail(e.getMessage());
