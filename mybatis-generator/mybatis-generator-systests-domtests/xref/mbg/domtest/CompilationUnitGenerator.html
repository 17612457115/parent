16  package mbg.domtest;
17  
18  import java.util.List;
19  
20  import org.mybatis.generator.api.dom.java.CompilationUnit;
21  
22  /**
23   * Implementors of this interface will be called during the build.  The expectation is that
24   * implementors will create hierarchies of classes/interfaces/enums of varying complexity.
25   * After they are created, the build will attempt to compile everything that was generated.
26   * If any code produced by the Java code generator is incorrect, the build should fail
27   * with compile errors in the generated objects.
28   * 
29   * Use the IgnoreDomTest annotation to skip a generator class (for example,
30   * when doing TDD to fix an error in the code generator and you want to commit the
31   * test before committing the fix)
32   * 
33   */
34  public interface CompilationUnitGenerator {
35      List<CompilationUnit> generate();
36  }
